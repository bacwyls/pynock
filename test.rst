Nock
====

>>> from nock import *

>>> l(1)
(1, 0)
>>> l(1, 2)
(1, 2)
>>> l(1, 2, 3)
(1, (2, 3))
>>> l(1, 2, 3, 4)
(1, (2, (3, 4)))

>>> wut(1)
1
>>> wut((1, 2))
0

>>> lus(1)
2

(1, 2)

>>> tis((1, 1))

>>> tis((1, 2))
1

>>> slot((1, 5))
5

5

(4, 5)
>>> slot((3, l(l(4, 5), l(6, 14, 15))))
(6, (14, 15))
>>> slot((7, l(l(4, 5), l(6, 14, 15))))
(14, 15)

>>> nock(l(l(l(4, 5), l(6, 14, 15)), 0, 7))
